Full E-Commerce Walkthrough

🎯 What You'll Build

This comprehensive tutorial walks you through building a complete e-commerce API using this framework. You'll create a fully functional online store with:

  • Product Management: List, view, create, and update products
  • Order Processing: Create orders with multiple items, calculate totals, manage inventory
  • User System: Registration, authentication, and user profiles
  • Admin Panel: Protected routes for product management

This tutorial combines all concepts from the previous documentation pages into one real-world application.

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- PRODUCTS TABLE
CREATE TABLE IF NOT EXISTS products (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    category_id INT UNSIGNED,
    name VARCHAR(255) NOT NULL,
    sku VARCHAR(100) UNIQUE NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    stock INT NOT NULL DEFAULT 0,
    status ENUM('active', 'inactive', 'out_of_stock') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_sku (sku),

    INDEX idx_status (status),
    FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- ORDERS TABLE
CREATE TABLE IF NOT EXISTS orders (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    user_id INT UNSIGNED NOT NULL,
    total_amount DECIMAL(10,2) NOT NULL,
    payment_method ENUM('card', 'cash', 'upi') NOT NULL,
    status ENUM('pending', 'confirmed', 'cancelled', 'completed') DEFAULT 'pending',

    cancellation_reason TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_user_id (user_id),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- ORDER ITEMS TABLE
CREATE TABLE IF NOT EXISTS order_items (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    order_id INT UNSIGNED NOT NULL,
    product_id INT UNSIGNED NOT NULL,
    product_name VARCHAR(255) NOT NULL,
    quantity INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_order_id (order_id),
    INDEX idx_product_id (product_id),
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

// File: src/api/repositories/ProductRepository.php

namespace App\Repositories;

use App\Core\Database;
use App\Core\Exceptions\NotFoundException;

class ProductRepository {
    private Database $db;

    public function __construct(Database $db) {
        $this->db = $db;
    }

    /**
     * Get all active products with optional filters
     */
    public function findAll(array $filters = []): array {
        $query = "SELECT 
                        p.id, p.name, p.sku, p.description, p.price, 
                        p.stock, p.status, p.created_at,
                        c.name as category_name, c.slug as category_slug
                    FROM products p
                    LEFT JOIN categories c ON p.category_id = c.id
                    WHERE p.status = :status";

        $params = ['status' => 'active'];

        if (!empty($filters['category'])) {
            $query .= " AND c.slug = :category";
            $params['category'] = $filters['category'];
        }

        $query .= " ORDER BY p.created_at DESC LIMIT :limit";
        $params['limit'] = $filters['limit'] ?? 50;

        return $this->db->select($query, $params);
    }

    /**
     * Find product by ID
     */
    public function findById(int $id): ?array {
        $query = "SELECT 
                        p.*, 
                        c.name as category_name, c.slug as category_slug
                    FROM products p
                    LEFT JOIN categories c ON p.category_id = c.id
                    WHERE p.id = :id";

        return $this->db->first($query, ['id' => $id]);
    }

    /**
     * Create new product
     */
    public function create(array $data): int {
        $query = "INSERT INTO products 
                      (category_id, name, sku, description, price, stock, status)
                   VALUES 
                      (:category_id, :name, :sku, :description, :price, :stock, :status)";

        $result = $this->db->insert($query, [
            'category_id' => $data['category_id'] ?? null,
            'name' => $data['name'],
            'sku' => $data['sku'],
            'description' => $data['description'] ?? null,
            'price' => $data['price'],
            'stock' => $data['stock'] ?? 0,
            'status' => $data['status'] ?? 'active'
        ]);

        return (int) $result['id'];
    }

    /**
     * Update product
     */
    public function update(int $id, array $data): bool {
        $setClauses = [];
        $params = ['id' => $id];

        foreach ($data as $key => $value) {
            $setClauses[] = "{$key} = :$key";
            $params[$key] = $value;
        }

        $query = "UPDATE products SET " . implode(', ', $setClauses) . " WHERE id = :id";
        $result = $this->db->update($query, $params);

        return $result['affectedRows'] > 0;
    }

    /**
     * Reduce product stock (for orders)
     */
    public function reduceStock(int $productId, int $quantity): bool {
        $query = "UPDATE products 
                      SET stock = stock - :quantity 
                      WHERE id = :id AND stock >= :quantity";

        $result = $this->db->update($query, [
            'id' => $productId,
            'quantity' => $quantity
        ]);

        return $result['affectedRows'] > 0;
    }

    /**
     * Restore product stock (for cancelled orders)
     */
    public function restoreStock(int $productId, int $quantity): bool {
        $query = "UPDATE products 
                      SET stock = stock + :quantity 
                      WHERE id = :id";

        $result = $this->db->update($query, [
            'id' => $productId,
            'quantity' => $quantity
        ]);

        return $result['affectedRows'] > 0;
    }
}

namespace App\Repositories;

use App\Core\Database;

class OrderRepository {
    private Database $db;

    public function __construct(Database $db) {
        $this->db = $db;
    }

    /**
     * Create new order
     */
    public function create(array $data): int {
        $query = "INSERT INTO orders 
                      (user_id, total_amount, payment_method, status, notes)
                   VALUES 
                      (:user_id, :total_amount, :payment_method, :status, :notes)";

        $result = $this->db->insert($query, [
            'user_id' => $data['user_id'],
            'total_amount' => $data['total_amount'],
            'payment_method' => $data['payment_method'],
            'status' => $data['status'] ?? 'pending',
            'notes' => $data['notes'] ?? null
        ]);

        return (int) $result['id'];
    }

    /**
     * Add order item
     */
    public function addItem(array $data): int {
        $query = "INSERT INTO order_items 
                      (order_id, product_id, product_name, quantity, price, subtotal)
                   VALUES 
                      (:order_id, :product_id, :product_name, :quantity, :price, :subtotal)";

        $result = $this->db->insert($query, $data);
        return (int) $result['id'];
    }

    /**
     * Find order by ID with items
     */
    public function findById(int $orderId): ?array {
        $query = "SELECT * FROM orders WHERE id = :id";
        $order = $this->db->first($query, ['id' => $orderId]);

        if (!$order) {
            return null;
        }

        // Get order items
        $order['items'] = $this->getOrderItems($orderId);

        return $order;
    }

    /**
     * Get order items
     */
    public function getOrderItems(int $orderId): array {
        $query = "SELECT * FROM order_items WHERE order_id = :order_id";
        return $this->db->select($query, ['order_id' => $orderId]);
    }

    /**
     * Update order status
     */
    public function updateStatus(int $orderId, string $status, ?string $reason = null): bool {
        $query = "UPDATE orders 
                      SET status = :status, 
                          cancellation_reason = :reason,
                          cancelled_at = " . ($status === 'cancelled' ? 'NOW()' : 'NULL') . "
                      WHERE id = :id";

        $result = $this->db->update($query, [
            'id' => $orderId,
            'status' => $status,
            'reason' => $reason
        ]);

        return $result['affectedRows'] > 0;
    }
}

// File: src/api/services/OrderService.php

namespace App\Services;

use App\Core\Database;
use App\Repositories\OrderRepository;
use App\Repositories\ProductRepository;
use App\Core\Exceptions\ValidationException;
use App\Core\Exceptions\NotFoundException;

class OrderService {
    private Database $db;
    private OrderRepository $orderRepo;
    private ProductRepository $productRepo;

    public function __construct(Database $db) {
        $this->db = $db;
        $this->orderRepo = new OrderRepository($db);
        $this->productRepo = new ProductRepository($db);
    }

    /**
     * Create order with items (uses transaction)
     */
    public function createOrder(array $data): array {
        // Validate items
        if (empty($data['items'])) {
            throw new ValidationException('Order must have at least one item');
        }

        // Use transaction for order creation
        return $this->db->transaction(function() use ($data) {
            // Calculate total and validate products
            $totalAmount = 0;
            $validatedItems = [];

            foreach ($data['items'] as $item) {
                $product = $this->productRepo->findById($item['product_id']);

                if (!$product) {
                    throw new NotFoundException("Product {$item['product_id']} not found");
                }

                if ($product['stock'] < $item['quantity']) {
                    throw new ValidationException("Insufficient stock for {$product['name']}");
                }

                $subtotal = $product['price'] * $item['quantity'];
                $totalAmount += $subtotal;

                $validatedItems[] = [
                    'product_id' => $product['id'],
                    'product_name' => $product['name'],
                    'quantity' => $item['quantity'],
                    'price' => $product['price'],
                    'subtotal' => $subtotal
                ];
            }

            // Create order
            $orderId = $this->orderRepo->create([
                'user_id' => $data['user_id'],
                'total_amount' => $totalAmount,
                'payment_method' => $data['payment_method'],
                'status' => 'pending',
                'notes' => $data['notes'] ?? null
            ]);

            // Add items and reduce stock
            foreach ($validatedItems as $item) {
                $item['order_id'] = $orderId;
                $this->orderRepo->addItem($item);

                // Reduce product stock
                $this->productRepo->reduceStock(
                    $item['product_id'], 
                    $item['quantity']
                );
            }

            return [
                'order_id' => $orderId,
                'total_amount' => $totalAmount,
                'items_count' => count($validatedItems)
            ];
        });
    }

    /**
     * Cancel order and restore stock
     */
    public function cancelOrder(int $orderId, string $reason): bool {
        $order = $this->orderRepo->findById($orderId);

        if (!$order) {
            throw new NotFoundException("Order {$orderId} not found");
        }

        if ($order['status'] === 'cancelled') {
            throw new ValidationException('Order is already cancelled');
        }

        // Use transaction
        return $this->db->transaction(function() use ($orderId, $reason, $order) {
            // Restore stock for each item
            foreach ($order['items'] as $item) {
                $this->productRepo->restoreStock(
                    $item['product_id'], 
                    $item['quantity']
                );
            }

            // Update order status
            return $this->orderRepo->updateStatus($orderId, 'cancelled', $reason);
        });
    }

    /**
     * Get order details
     */
    public function getOrderDetails(int $orderId): array {
        $order = $this->orderRepo->findById($orderId);

        if (!$order) {
            throw new NotFoundException("Order {$orderId} not found");
        }

        return $order;
    }
}

// File: src/api/controllers/ProductController.php

namespace App\Controllers;

use App\Core\BaseController;
use App\Repositories\ProductRepository;
use App\Core\Exceptions\NotFoundException;
use App\Core\Exceptions\ValidationException;

class ProductController extends BaseController {
    private ProductRepository $productRepo;

    public function __construct() {
        parent::__construct();
        $this->productRepo = new ProductRepository($this->db);
    }

    /**
     * GET /products/list
     * List all products with optional filtering
     */
    public function list() {
        try {
            $filters = [
                'category' => $_GET['category'] ?? null,
                'limit' => (int) ($_GET['limit'] ?? 50)
            ];

            $products = $this->productRepo->findAll($filters);

            $this->sendSuccess('Products retrieved', [
                'products' => $products,
                'count' => count($products)
            ]);

        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }

    /**
     * GET /products/details?product_id=1
     * Get product details
     */
    public function details() {
        try {
            $productId = $_GET['product_id'] ?? null;

            if (!$productId) {
                $this->sendError('Product ID is required', 400);
                return;
            }

            $product = $this->productRepo->findById((int) $productId);

            if (!$product) {
                throw new NotFoundException("Product {$productId} not found");
            }

            $this->sendSuccess('Product found', $product);

        } catch (NotFoundException $e) {
            throw $e; // Let ExceptionHandler format it
        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }

    /**
     * POST /products/create
     * Create new product (Admin only)
     */
    public function create() {
        try {
            $data = $this->getRequestData();

            // Validate
            $errors = $this->validateRequest([
                'name' => 'required',
                'sku' => 'required',
                'price' => 'required|numeric',
                'stock' => 'required|integer'
            ]);

            if (!empty($errors)) {
                throw new ValidationException('Validation failed', $errors);
            }

            $productId = $this->productRepo->create($data);

            $this->sendSuccess('Product created', [
                'product_id' => $productId
            ]);

        } catch (ValidationException $e) {
            throw $e;
        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }
}

namespace App\Controllers;

use App\Core\BaseController;
use App\Services\OrderService;
use App\Core\Exceptions\ValidationException;

class OrderController extends BaseController {
    private OrderService $orderService;

    public function __construct() {
        parent::__construct();
        $this->orderService = new OrderService($this->db);
    }

    /**
     * POST /orders/create
     * Create new order
     */
    public function create() {
        try {
            $data = $this->getRequestData();

            // Validate
            $errors = $this->validateRequest([
                'user_id' => 'required|integer',
                'payment_method' => 'required',
                'items' => 'required|array'
            ]);

            if (!empty($errors)) {
                throw new ValidationException('Validation failed', $errors);
            }

            $result = $this->orderService->createOrder($data);

            $this->sendSuccess('Order created successfully', $result);

        } catch (ValidationException $e) {
            throw $e;
        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }

    /**
     * GET /orders/details?order_id=1
     * Get order details
     */
    public function details() {
        try {
            $orderId = $_GET['order_id'] ?? null;

            if (!$orderId) {
                $this->sendError('Order ID is required', 400);
                return;
            }

            $order = $this->orderService->getOrderDetails((int) $orderId);

            $this->sendSuccess('Order found', $order);

        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }

    /**
     * POST /orders/cancel
     * Cancel order
     */
    public function cancel() {
        try {
            $data = $this->getRequestData();

            $orderId = $data['order_id'] ?? null;
            $reason = $data['reason'] ?? 'Customer request';

            if (!$orderId) {
                $this->sendError('Order ID is required', 400);
                return;
            }

            $this->orderService->cancelOrder((int) $orderId, $reason);

            $this->sendSuccess('Order cancelled successfully');

        } catch (\Exception $e) {
            $this->sendServerError($e->getMessage());
        }
    }
}
